   32 {-|
   33   Room
   34   Represents a room with key characteristics such as ID, capacity, accessibility features, 
   35   and a list of scheduled bookings.
   36 
   37   Fields:
   38     roomID (String)                         - Unique identifier for the room
   39     capacity (Int)                          - Maximum allowed group size
   40     wheelchairAccess (Bool)                 - Indicates if the room supports accessibility needs
   41     projector (Bool)                        - Indicates if the room has a projector
   42     computer (Bool)                         - Indicates if the room has a computer
   43     floorLevel (Int)                        - The floor number of the room
   44     schedule ([(UTCTime, UTCTime, String)]) - Current list of bookings with time range and group ID
   45 -}
   46 data Room = Room
   47   { roomID           :: String
   48   , capacity         :: Int
   49   , wheelchairAccess :: Bool
   50   , projector        :: Bool
   51   , computer         :: Bool
   52   , floorLevel       :: Int

  140 {- addSchedule
  141    Adds a new booking to the beginning of the room’s schedule.
  142 
  143    Parameters:
  144      Room - the room to update
  145      (UTCTime, UTCTime, String) - booking entry
  146 
  147    Return Value:
  148      Room - updated room with the new booking prepended
  149 -}
  150 addSchedule :: Room -> (UTCTime, UTCTime, String) -> Room
  151 addSchedule room newEntry = room { schedule = newEntry : schedule room }
