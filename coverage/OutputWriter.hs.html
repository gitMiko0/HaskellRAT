    1 {-
    2 Module Name: OutputWriter.hs
    3 Project Name: Room Assignment Tool (Functional/Haskell Solution)
    4 File Purpose: Responsible for formatting and exporting the final assignment solution to human-readable output or CSV.
    5 
    6 Module Summary:
    7 This module provides functionality to format a solution (list of assignments) either as a multiline string for console
    8 display, or as a CSV file suitable for output submission. It uses standard time formatting for readability.
    9 
   10 Key Functions:
   11 - `formatSolution` - generates a readable string version of the assignments
   12 - `writeCSV` - writes assignments to a CSV file
   13 - `formatAsCSV` - formats a single assignment into a CSV row
   14 
   15 Dependencies:
   16 - Data.Time (for formatting timestamps)
   17 - System.IO (for file output)
   18 - Solver.hs (uses the `Solution` type alias)
   19 
   20 Known/Suspected Errors:
   21 - None known at this time
   22 -}
   23 
   24 module OutputWriter (
   25   formatSolution,
   26   writeCSV
   27 ) where
   28 
   29 import Data.Time (UTCTime, formatTime, defaultTimeLocale)
   30 import System.IO
   31 import Solver (Solution)
   32 
   33 {-
   34 formatSolution
   35   Converts a list of assignments into a multiline string with a readable format.
   36 
   37 Parameters:
   38   Solution - A list of tuples (GroupID, RoomID, StartTime, EndTime)
   39 
   40 Return Value:
   41   String - Formatted string with each assignment on its own line
   42 -}
   43 formatSolution :: Solution -> String
   44 formatSolution = unlines . map formatLine

   45   where

   46     fmt t = formatTime defaultTimeLocale "%Y-%m-%d %H:%M" t

   47     formatLine (group, room, start, end) =

   48       group ++ " -> " ++ room ++ " : " ++ fmt start ++ " - " ++ fmt end
   49 
   50 {-
   51 writeCSV
   52   Writes the solution to a CSV file with the format:
   53   GroupID,RoomID,StartTime,EndTime
   54 
   55 Parameters:
   56   file :: FilePath - The output CSV file path
   57   solution :: Solution - The list of assignments to export
   58 
   59 Return Value:
   60   IO () - Performs file writing as a side effect
   61 -}
   62 writeCSV :: FilePath -> Solution -> IO ()
   63 writeCSV file solution = do

   64   writeFile file (unlines (map formatAsCSV solution))
   65 
   66 {-
   67 formatAsCSV
   68   Formats a single assignment tuple as a CSV line.
   69 
   70 Parameters:
   71   (String, String, UTCTime, UTCTime) - (GroupID, RoomID, Start, End)
   72 
   73 Return Value:
   74   String - Comma-separated string suitable for CSV export
   75 -}
   76 formatAsCSV :: (String, String, UTCTime, UTCTime) -> String
   77 formatAsCSV (group, room, start, end) =

   78   let fmt = formatTime defaultTimeLocale "%Y-%m-%d %H:%M"

   79   in group ++ "," ++ room ++ "," ++ fmt start ++ "," ++ fmt end
