    1 {-
    2 Module Name: OutputWriter.hs
    3 Project Name: Room Assignment Tool (Functional/Haskell Solution)
    4 File Purpose: Responsible for formatting and exporting the final assignment solution to human-readable output or CSV.
    5 
    6 Module Summary:
    7 This module provides functionality to format a solution (list of assignments) either as a multiline string for console
    8 display, or as a CSV file suitable for output submission. It uses standard time formatting for readability.
    9 
   10 Key Functions:
   11 - `formatSolution` - generates a readable string version of the assignments
   12 - `writeCSV` - writes assignments to a CSV file
   13 - `formatAsCSV` - formats a single assignment into a CSV row
   14 
   15 Dependencies:
   16 - Data.Time (for formatting timestamps)
   17 - System.IO (for file output)
   18 - Solver.hs (uses the `Solution` type alias)
   19 
   20 Known/Suspected Errors:
   21 - None known at this time
   22 -}
   23 
   24 module OutputWriter (
   25   formatSolution,
   26   writeCSV
   27 ) where
   28 
   29 import Data.Time (UTCTime, formatTime, defaultTimeLocale)
   30 import Data.List (intercalate)
   31 
   32 import System.IO
   33 import Solver (Solution)
   34 
   35 {-
   36 formatSolution
   37   Converts a list of assignments into a multiline string with a readable format.
   38 
   39 Parameters:
   40   Solution - A list of tuples (GroupID, RoomID, StartTime, EndTime)
   41 
   42 Return Value:
   43   String - Formatted string with each assignment on its own line
   44 -}
   45 formatSolution :: Solution -> String
   46 formatSolution = unlines . map formatLine

   47   where

   48     fmt t = formatTime defaultTimeLocale "%Y-%m-%d %H:%M" t

   49     formatLine (group, room, start, end) =

   50       group ++ " -> " ++ room ++ " : " ++ fmt start ++ " - " ++ fmt end
   51 
   52 {-|
   53 writeCSV
   54   Writes the formatted solution as a CSV file with a header line.
   55 
   56 Parameters:
   57   FilePath - the target CSV file path
   58   Solution - the list of group-room-time assignments
   59 
   60 Return Value:
   61   IO () - performs the write side-effect
   62 
   63 Exceptions:
   64   May raise I/O exceptions if the file path is invalid or unwritable.
   65 -}
   66 writeCSV :: FilePath -> Solution -> IO ()
   67 writeCSV path sol = writeFile path $ unlines $

   68   ["GroupID,RoomID,Start,End"] ++ map toCSV sol --header

   69   where

   70     toCSV (gid, rid, start, end) =

   71       intercalate "," [gid, rid, fmt start, fmt end]  --join separated with commas

   72     fmt = formatTime defaultTimeLocale "%Y-%m-%d %H:%M:%S"
   73 
   74 {-
   75 formatAsCSV
   76   Formats a single assignment tuple as a CSV line.
   77 
   78 Parameters:
   79   (String, String, UTCTime, UTCTime) - (GroupID, RoomID, Start, End)
   80 
   81 Return Value:
   82   String - Comma-separated string suitable for CSV export
   83 -}
   84 formatAsCSV :: (String, String, UTCTime, UTCTime) -> String
   85 formatAsCSV (group, room, start, end) =

   86   let fmt = formatTime defaultTimeLocale "%Y-%m-%d %H:%M"

   87   in group ++ "," ++ room ++ "," ++ fmt start ++ "," ++ fmt end
