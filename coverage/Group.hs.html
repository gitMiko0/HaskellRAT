   33 -- | Group
   34 -- Represents a participant group with scheduling details and equipment requirements.
   35 data Group = Group
   36   { groupID         :: String        -- ^ Unique identifier for the group
   37   , size            :: Int           -- ^ Number of participants in the group
   38   , startTime       :: UTCTime       -- ^ Group's requested start time
   39   , endTime         :: UTCTime       -- ^ Group's requested end time
   40   , projectorNeed   :: Bool          -- ^ True if the group needs a projector
   41   , computerNeed    :: Bool          -- ^ True if the group needs a computer
   42   , wheelchairNeed  :: Bool          -- ^ True if accessibility is required
   43   , floorPreference :: Int           -- ^ Preferred floor (-1 = no preference)
   44   } deriving (Show, Eq)
   45 
   46 -- | getGroupID
