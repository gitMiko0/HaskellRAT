   37 parseTests :: Test
   38 parseTests = TestList

   39   [ TestCase $ do

   40       let input = "R1,10,TRUE,FALSE,TRUE,2"

   41           r = parseRoom input

   42       assertEqual "Room ID" "R1" (getRoomID r)

   43 

   44   , TestCase $ do

   45       let input = "G1,5,FALSE,FALSE,TRUE,1,2023-01-01 10:00,2023-01-01 11:00"

   46           g = parseGroup input

   47       assertEqual "Group ID" "G1" (getGroupID g)

   48       assertEqual "Start" (getStartTime g) (getStartTime (parseGroup input))

   49 

   50   , TestCase $ do

   51       result <- try (evaluate (parseGroup "G1,5,FALSE,FALSE,TRUE,1,invalid-date,2023-01-01 11:00")) :: IO (Either ErrorCall Group)

   52       case result of

   53         Left (ErrorCall msg) -> assertBool "Invalid date error" ("Invalid date format" `isInfixOf` msg)

   54         _ -> assertFailure "Expected error on bad date"

   55 

   56   , TestCase $ do

   57       result <- try (evaluate (parseRoom "R1,not-a-number,TRUE,TRUE,FALSE,1")) :: IO (Either ErrorCall Room)

   58       case result of

   59         Left (ErrorCall msg) -> assertBool "Invalid int error" ("Invalid capacity" `isInfixOf` msg)

   60         _ -> assertFailure "Expected error on bad int"

   61 

   62   , TestCase $ do

   63       result <- try (evaluate (parseRoom "R1,10,maybe,TRUE,FALSE,1")) :: IO (Either ErrorCall Room)

   64       case result of

   65         Left (ErrorCall msg) -> assertBool "Invalid bool error" ("Invalid boolean for WheelchairAccess" `isInfixOf` msg)

   66         _ -> assertFailure "Expected error on bad boolean"

   67 

   68     -- The following tests ensure that the loading function is correctly populating the data structures
