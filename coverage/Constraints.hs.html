   45 {-|
   46 isValidAssignment  
   47   Aggregates all constraint checks and returns True only if all are satisfied
   48   for assigning a group to a room, including time conflict detection with a
   49   user-defined buffer between bookings.
   50 
   51 Parameters:
   52   Group             - the group being considered  
   53   Room              - the room being considered  
   54   NominalDiffTime   - required time gap between bookings (in seconds)
   55 
   56 Return Value:
   57   Bool - True if all constraints pass
   58 -}
   59 isValidAssignment :: Group -> Room -> NominalDiffTime -> Bool
   60 isValidAssignment group room gap =

   61   checkFloorPreference group room &&

   62   checkRoomCapacity group room &&

   63   checkWheelchairAccess group room &&

   64   checkEquipment group room &&

   65   checkScheduleConflict group room gap
   66   
   67 {-|
   68 checkFloorPreference  
   69   Verifies whether the group's floor preference matches the room's floor level.
   70   If the group has no preference (-1), this check always passes.
   71 
   72 Parameters:
   73   Group - group requesting room  
   74   Room  - candidate room for assignment
   75 
   76 Return Value:
   77   Bool - True if floor level is acceptable
   78 -}
   79 checkFloorPreference :: Group -> Room -> Bool
   80 checkFloorPreference group room =

   81   let pref = getFloorPreference group

   82   in pref == -1 || pref == getFloorLevel room
   83 
   84 {-|
   85 checkRoomCapacity  
   86   Ensures that the room has enough capacity for the group's size.
   87 
   88 Parameters:
   89   Group - group requesting room  
   90   Room  - candidate room
   91 
   92 Return Value:
   93   Bool - True if room can hold the group
   94 -}
   95 checkRoomCapacity :: Group -> Room -> Bool
   96 checkRoomCapacity group room = getCapacity room >= getSize group
   97 
   98 {-|
   99 checkWheelchairAccess  
  100   Checks if the room satisfies the group's wheelchair accessibility requirement.
  101 
  102 Parameters:
  103   Group - group requesting room  
  104   Room  - candidate room
  105 
  106 Return Value:
  107   Bool - True if wheelchair needs are satisfied
  108 -}
  109 checkWheelchairAccess :: Group -> Room -> Bool
  110 checkWheelchairAccess group room =

  111   not (needsWheelchair group) || hasWheelchairAccess room
  112 
  113 {-|
  114 checkEquipment  
  115   Verifies that the room contains all equipment the group requires.
  116 
  117 Parameters:
  118   Group - group requesting room  
  119   Room  - candidate room
  120 
  121 Return Value:
  122   Bool - True if all required equipment is available
  123 -}
  124 checkEquipment :: Group -> Room -> Bool
  125 checkEquipment group room =

  126   (not (needsProjector group) || hasProjector room) &&

  127   (not (needsComputer group) || hasComputer room)
  128 
  129 {-|
  130 checkScheduleConflict  
  131   Ensures that the group's schedule does not conflict with any existing bookings
  132   in the room, accounting for a user-defined time gap between assignments.
  133 
  134 Parameters:
  135   Group             - group to check  
  136   Room              - room whose schedule will be compared  
  137   NominalDiffTime   - required time gap between bookings (converted from user-defined minutes)
  138 
  139 Return Value:
  140   Bool - True if there are no overlaps, including the required buffer
  141 -}
  142 checkScheduleConflict :: Group -> Room -> NominalDiffTime -> Bool
  143 checkScheduleConflict group room gap =

  144   all (\existing -> checkTimeOverlap (getStartTime group, getEndTime group) existing gap)

  145       (getSchedule room)
  146 
  147 {-|
  148 checkTimeOverlap  
  149   Compares a new time block with an existing one and determines if they overlap,
  150   using a required buffer time between them.
  151 
  152 Parameters:
  153   (UTCTime, UTCTime)             - start and end of the new booking  
  154   (UTCTime, UTCTime, String)     - existing booking (with group ID)  
  155   NominalDiffTime                - required time gap in seconds
  156 
  157 Return Value:
  158   Bool - True if no overlap exists with the specified buffer applied
  159 -}
  160 checkTimeOverlap :: (UTCTime, UTCTime) -> (UTCTime, UTCTime, String) -> NominalDiffTime -> Bool
  161 checkTimeOverlap (newStart, newEnd) (existingStart, existingEnd, _) gap =

  162   addUTCTime gap newEnd <= existingStart || newStart >= addUTCTime gap existingEnd
