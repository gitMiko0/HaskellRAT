   30 solverTests = TestList

   31   [ -- Valid single-room assignment

   32     TestCase $ do

   33       let g1 = group "10:00" "11:00" 5 False False False 1 "2023-01-01" "G1"

   44           r1 = room "R1" 10 True True True 1

   45           r2 = room "R2" 10 True True True 1

   46       case assignGroups [g1, g2] [r1, r2] testGap of

   49
