   42 {-
   43 loadCSV
   44   Loads a CSV file and applies a parsing function to each non-header line.
   45 
   46 Parameters:
   47   parseFunc :: String -> a
   48     A function that parses a line of CSV into a desired type (Room or Group)
   49 
   50   file :: FilePath
   51     Path to the input CSV file
   52 
   53 Return Value:
   54   IO [a] - List of parsed values of type `a`
   55 -}
   56 loadCSV :: (String -> a) -> FilePath -> IO [a]
   57 loadCSV parseFunc file = do

   58   contents <- readFile file

   59   let lineList = drop 1 (lines contents)  -- Skip the header

   60   return $ map parseFunc lineList
   61 
   62 {-
   63 parseCSVLine
   64   Splits a CSV line into individual, trimmed string values.
   65 
   66 Parameters:
   67   String - A raw CSV line
   68 
   69 Return Value:
   70   [String] - Cleaned list of values from the line
   71 -}
   72 parseCSVLine :: String -> [String]
   73 parseCSVLine = map trim . splitOn ","
   74 
   75 {-|
   76 parseBoolOrFail  
   77   Strictly parses a string into a Bool. Only "TRUE" or "FALSE" (case-insensitive) are accepted.
   78   Raises an error for any other value.
   79 
   80 Parameters:
   81   fieldName :: String - The name of the field being parsed (used in error message)
   82   raw       :: String - The raw input string
   83 
   84 Return Value:
   85   Bool - Parsed boolean value
   86 
   87 Exceptions:
   88   Raises "Invalid boolean for <fieldName>: <value>" if the input is invalid.
   89 -}
   90 
   91 parseBoolOrFail :: String -> String -> Bool
   92 parseBoolOrFail fieldName raw =

   93   case map toUpper (trim raw) of

   94     "TRUE"  -> True

   95     "FALSE" -> False

   96     other   -> error $ "Invalid boolean for " ++ fieldName ++ ": " ++ show other

   99 {-
  100 parseDateTime
  101   Parses a string in the format "YYYY-MM-DD HH:MM" to UTCTime.
  102 
  103 Parameters:
  104   String - A datetime string from CSV
  105 
  106 Return Value:
  107   Maybe UTCTime - Parsed time or Nothing on failure
  108 -}
  109 parseDateTime :: String -> Maybe UTCTime
  110 parseDateTime dateStr =

  111   parseTimeM True defaultTimeLocale "%Y-%m-%d %H:%M" (trim dateStr)
  112 
  113 {-
  114 trim
  115   Removes leading and trailing whitespace from a string.
  116 
  117 Parameters:
  118   String - Input string
  119 
  120 Return Value:
  121   String - Trimmed output
  122 -}
  123 trim :: String -> String
  124 trim = f . f

  125   where f = reverse . dropWhile (== ' ')

  127 {-|
  128 parseRoom  
  129   Converts a line from the room CSV file into a Room data structure.
  130 
  131 Parameters:
  132   String - CSV line containing RoomID, Capacity, WheelchairAccess, Projector, Computer, FloorLevel
  133 
  134 Return Value:
  135   Room - Parsed Room record
  136 
  137 Exceptions:
  138   Raises errors for:
  139     - Invalid capacity
  140     - Invalid floor level
  141     - Invalid boolean values
  142 
  143 Note:
  144   Uses Bang Patterns to force evaluation of parsed values to ensure that all malformed input
  145   raises errors at parse-time rather than later due to Haskell's lazy evaluation.
  146 -}
  147 
  148 parseRoom :: String -> Room
  149 parseRoom line =

  150   let [rid, cap, wca, proj, comp, floor] = parseCSVLine line

  151       !capacity   = readOrFail "Invalid capacity" cap

  152       !level      = readOrFail "Invalid floor level" floor

  153       !wheelchair = parseBoolOrFail "WheelchairAccess" wca

  154       !projector  = parseBoolOrFail "Projector" proj

  155       !computer   = parseBoolOrFail "Computer" comp

  156   in Room rid capacity wheelchair projector computer level []
  157 
  158 
  159 {-
  160 parseGroup
  161   Converts a line from the group CSV file into a Group data structure.
  162 
  163 Parameters:
  164   String - CSV line containing GroupID, Size, WheelchairAccess, Projector, Computer, FloorPreference, Start, End
  165 
  166 Return Value:
  167   Group - Parsed Group record
  168 
  169 Exceptions:
  170   Raises an error on malformed date strings or integer parsing failure.
  171 -}
  172 parseGroup :: String -> Group
  173 parseGroup line =

  174   let [gid, size, wca, proj, comp, floor, start, end] = parseCSVLine line

  175       !pref       = readOrFail "Invalid floor preference" floor

  176       !groupSize  = readOrFail "Invalid size" size

  177       !wheelchair = parseBoolOrFail "WheelchairAccess" wca

  178       !projector  = parseBoolOrFail "Projector" proj

  179       !computer   = parseBoolOrFail "Computer" comp

  180       startTime   = parseDateTime start

  181       endTime     = parseDateTime end

  182   in case (startTime, endTime) of

  183       (Just st, Just et) ->

  184         Group gid groupSize st et projector computer wheelchair pref

  185       _ -> error $ "Invalid date format: " ++ start ++ " / " ++ end

  188 {-
  189 readIntOrFail
  190   Attempts to parse a string as an integer or raises a custom error message.
  191 
  192 Parameters:
  193   errMsg :: String - Message to show if parsing fails
  194   s      :: String - Input string to parse
  195 
  196 Return Value:
  197   Int - Parsed integer or triggers error
  198 -}
  199 readOrFail :: String -> String -> Int
  200 readOrFail errMsg s = fromMaybe (error errMsg) (readMaybe (trim s))

  202 runWithGap :: FilePath -> FilePath -> NominalDiffTime -> IO ()
  203 runWithGap roomsFile groupsFile gap = do

  204   rooms <- loadCSV parseRoom roomsFile

  205   groups <- loadCSV parseGroup groupsFile

  206   case assignRooms groups rooms gap of

  207     Just sol -> do

  208       putStrLn "Room Assignments:"

  209       putStrLn (formatSolution sol)

  210       writeCSV "assignments.csv" sol

  211     Nothing -> putStrLn "Error: Constraints cannot be satisfied with the provided input."
