   35 -- | A Solution is a list of successful group-room-time assignments
   36 type Solution = [(String, String, UTCTime, UTCTime)]
   37 
   38 -- | Assigns groups to rooms using backtracking.
   39 --   Returns updated list of rooms with scheduled bookings.
   40 assignGroups :: [Group] -> [Room] -> NominalDiffTime -> Maybe Solution
   41 assignGroups [] _ _ = Just []

   42 assignGroups (g:gs) rooms timeGap = tryRooms rooms

   43   where

   44     tryRooms [] = Nothing

   45     tryRooms (r:rs)

   46       | isValidAssignment g r timeGap =

   47           let updatedRoom = addSchedule r (getStartTime g, getEndTime g, getGroupID g)

   48               updatedRooms = replaceRoom updatedRoom rooms

   49           in case assignGroups gs updatedRooms timeGap of

   50                Just sol -> Just ((getGroupID g, getRoomID r, getStartTime g, getEndTime g) : sol)

   51                Nothing  -> tryRooms rs

   52       | otherwise = tryRooms rs
   53 
   54 
   55 -- | Replaces a room in the list with an updated version by matching RoomID
   56 replaceRoom :: Room -> [Room] -> [Room]
   57 replaceRoom updated = map (\r -> if getRoomID r == getRoomID updated then updated else r)
